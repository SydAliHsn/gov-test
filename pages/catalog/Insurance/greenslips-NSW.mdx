import { useSSG } from 'nextra/ssg';
import screenshot from '../../../lib/screenshot';
import ScreenshotImage from '../../../components/ScreenshotImage';

export const getStaticProps = () => {
const url = 'https://www.greenslips.nsw.gov.au/'

return screenshot(url).then((imageUrl) => ({
props: { ssg: {imageUrl} },
}))
};

export const Screenshot = () => {
const {imageUrl} = useSSG();

return <ScreenshotImage imageUrl={imageUrl}/>
}

# Greenslips NSW

### About

Green Slip Price Check is a quick and easy way to get a Green Slip quote comparison from all insurers. This tool is for NSW users.

### Operated by

State Insurance Regulatory Authority, NSW Government.
