import { useSSG } from 'nextra/ssg';
import screenshot from '../../../lib/screenshot';
import ScreenshotImage from '../../../components/ScreenshotImage';

export const getStaticProps = () => {
const url = 'https://headtohealth.gov.au/'

return screenshot(url).then((imageUrl) => ({
props: { ssg: {imageUrl} },
}))
};

export const Screenshot = () => {
const {imageUrl} = useSSG();

return <ScreenshotImage imageUrl={imageUrl}/>
}

# Head to health

### About

Head to Health is a free confidential service from the Australian Government. It connects you with the help and support you need to keep mentally healthy.

### Operated by

Department of Health
