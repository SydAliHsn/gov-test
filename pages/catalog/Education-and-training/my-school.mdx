import { useSSG } from 'nextra/ssg';
import screenshot from '../../../lib/screenshot';

export const getStaticProps = () => {
const url = 'https://www.myschool.edu.au/'

return screenshot(url).then((imageUrl) => ({
props: { ssg: {imageUrl} },
}))
};

export const ScreenshotImage = () => {
const {imageUrl} = useSSG();

return <img src={imageUrl} alt="screenshot"/>
}

# My School

### About

My School supports national transparency and accountability of Australia's schools by publishing nationally consistent school-level data about every school in Australia.

### Operated by

Australian Curriculum, Assessment and Reporting Authority (ACARA)
