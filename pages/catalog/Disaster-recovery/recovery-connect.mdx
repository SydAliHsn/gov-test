import { useSSG } from 'nextra/ssg';
import ImageLink from '../../../components/ImageLink';
import screenshot from '../../../lib/screenshot';

export const getStaticProps = () => {
const url = 'https://recovery.serviceconnect.gov.au/'

return screenshot(url).then((image) => ({
props: { ssg: {image} },
}))
};

export const Imgi = () => {

const {image} = useSSG();

return <ImageLink url={image} />
}
